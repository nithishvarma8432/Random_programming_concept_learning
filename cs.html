Python

If-Else Statements

if x > 10:
    print("x is greater than 10")
else:
    print("x is not greater than 10")

For Loops

for i in range(5):
    print(i)

While Loops

while x < 10:
    print(x)
    x += 1

JavaScript

If-Else Statements

if (x > 10) {
    console.log("x is greater than 10");
} else {
    console.log("x is not greater than 10");
}

For Loops

for (let i = 0; i < 5; i++) {
    console.log(i);
}

While Loops

while (x < 10) {
    console.log(x);
    x++;
}

Java

If-Else Statements

if (x > 10) {
    System.out.println("x is greater than 10");
} else {
    System.out.println("x is not greater than 10");
}

For Loops

for (int i = 0; i < 5; i++) {
    System.out.println(i);
}

While Loops

while (x < 10) {
    System.out.println(x);
    x++;
}
